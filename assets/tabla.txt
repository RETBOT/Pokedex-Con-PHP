CREATE TABLE favoritos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    numero VARCHAR(10),
    url VARCHAR(255),
    imagen VARCHAR(255),
    descripcion_version_x TEXT,
    descripcion_version_y TEXT,
    altura VARCHAR(10),
    categoria VARCHAR(50),
    peso VARCHAR(10),
    habilidad VARCHAR(50),
    sexo VARCHAR(50),
    tipo VARCHAR(255),
    debilidad VARCHAR(255),
    evoluciones VARCHAR(255),
    ps INT,
    ataque INT,
    defensa INT,
    ataque_especial INT,
    defensa_especial INT,
    velocidad INT
);
